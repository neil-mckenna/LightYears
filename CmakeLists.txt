cmake_minimum_required(VERSION 3.30.3)

project(LightYears
	VERSION 1.0.0
	LANGUAGES C CXX
)

set(CMAKE_CXX_STANARD 			17)
set(CMAKE_CXX_STANARD_REQUIRED 	ON)
set(CMAKE_CXX_EXTENSIONS 		OFF)

include(FetchContent)
set(SFML_LIBRARY_NAME SFML)
set(SFML_LIBRARY_INCLUDES
    sfml-graphics
    sfml-main
    sfml-system
    sfml-network
    sfml-audio
)

FetchContent_Declare(
	${SFML_LIBRARY_NAME}
	GIT_REPOSITORY https://github.com/SFML/SFML
	GIT_TAG 2.5.1
	GIT_SHALLOW FALSE
)

#FetchContent_MakeAvailable(${SFML_LIBRARY_NAME})
FetchContent_MakeAvailable(SFML)

# create a variable for title name. 
set(LIGHT_YEARS_GAME_TARGET_NAME LightYearsGame)

# this call this cmakelist beside the src files !!! inside the otehr cmakelist there is the target linkers
add_subdirectory(LightYears)
